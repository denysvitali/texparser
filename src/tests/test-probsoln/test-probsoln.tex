\begin{defproblem}[1][{2}]{diffsin}
Differentiate $f(x)=\sin(#1x)$.
\begin{onlysolution}
 \begin{solution}
 $f'(x) = #1\cos(#1)x$.
 \end{solution}
\end{onlysolution}
\end{defproblem}

\newproblem[1][{4}]{diffcos}{%
  \(f(x) = \cos(#1x)\)
}%
{%
  \(f'(x) = -#1\sin(#1x)\)
}

\newproblem*{essay}{An essay style question without an answer.
\lipsum[1]}

\begin{defproblem}{subparts}
Find $x$ in:
\begin{textenum}
\item $4+x=2$;
\item $3x=9$;
\item $x-1=4$.
\end{textenum}
\begin{onlysolution}
 \begin{solution}
  \begin{textenum}
  \item $x=-2$;
  \item $x=3$;
  \item $x=5$.
  \end{textenum}
 \end{solution}
\end{onlysolution}
\end{defproblem}
