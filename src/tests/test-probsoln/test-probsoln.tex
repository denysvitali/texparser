\begin{defproblem}[1][{2}]{diffsin}
Differentiate $f(x)=\sin(#1x)$.
\begin{onlysolution}
 \begin{solution}
 $f'(x) = #1\cos(#1)x$.
 \end{solution}
\end{onlysolution}
\end{defproblem}

\newproblem[1][{4}]{diffcos}{%
  \(f(x) = \cos(#1x)\)
}%

  \(f'(x) = -#1\sin(#1x)\)
}
